function initProgressiveStatisticsTab(options = {}) { const tabElement = document.getElementById(options.tabId || 'stats-tab'); const loadingElement = document.getElementById( options.loadingId || 'loadingMessageStats', ); const groupFilterElement = document.getElementById( options.groupFilterId || 'groupFilterStats', ); const resultElement = document.getElementById( options.resultId || 'statisticsResultStats', ); const errorElement = document.getElementById( options.errorId || 'errorMessageStats', ); if (!tabElement || !groupFilterElement || !resultElement) { console.error('統計頁初始化失敗：缺少必要 DOM 元素'); return; } let statsLoaded = false; let summaryStats = null; let statsDetailState = { groupName: '', page: 1, pageSize: 10, hasMore: false, items: [], allItems: [], totalItems: 0, version: '', generatedAt: '', }; function escapeHtmlStats(unsafe) { if (typeof unsafe !== 'string') return ''; return unsafe .replace(/&/g, '&') .replace(//g, '>') .replace(/"/g, '"') .replace(/'/g, '''); } function displayErrorStats(error) { if (loadingElement) { loadingElement.style.display = 'none'; } console.error('統計資料載入錯誤:', error); if (errorElement) { errorElement.textContent = '載入統計資料時發生錯誤：' + (error.message || '未知錯誤') + '. 請檢查主控台獲取更多資訊。'; } } function getMinutesSince(generatedAt) { if (!generatedAt) return '-'; const diffMs = Date.now() - new Date(generatedAt).getTime(); if (Number.isNaN(diffMs)) return '-'; return Math.max(Math.floor(diffMs / 60000), 0); } function renderSummaryStats(summary) { const groups = summary.groups || []; if (groups.length === 0) { resultElement.innerHTML = '

'; return; } const maxCount = Math.max( ...currentPageItems.map((item) => item.count || 0), 1, ); const minutesAgo = getMinutesSince(statsDetailState.generatedAt); let barsHtml = ''; currentPageItems.forEach((item) => { const count = item.count || 0; const width = Math.max(Math.round((count / maxCount) * 100), 2); barsHtml += `
